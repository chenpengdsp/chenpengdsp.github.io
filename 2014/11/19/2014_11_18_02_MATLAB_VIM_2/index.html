VIM 插件推荐

为了完善VIM的功能，我将介绍几个插件，包括Ctags, WinManager, pathogen, EasyMotion, Tabular。

Ctags 插件

  • Ctags 插件的作用

    Ctags 插件实现了函数之间的跳转功能，即当我们在代码中调用某个函数时，如果我们想进入该函数的定义位置，便可采用Ctags插件实现。

  • Ctags 插件的安装

    1. 到官方网站（http://ctags.sourceforge.net/ ）下载Ctags，选择“Source and binary for Windows 98/NT/2000/XP”版本，解压缩下载的文件，将ctags.exe文件放到 D:\Vim\vim74\ 文件夹中即可。

    1. Ctags 的工作依靠于tags文件，在Normal模式下，按下 F8 生成Tags文件。

    2. 在函数名位置处按下 Ctrl+]，进入该函数的定义位置。按下 Ctrl+T 返回。

NERDTree 插件

  • NERDTree 插件的作用

    NERDTree 插件配合 WinManager 插件，可以在 VIM 中显示文件浏览页面，从而实现在 VIM 浏览文件夹下不同文件，并切换编辑的功能

  • NERDTree 插件的安装

    1. 请到 VIM 的官方网站 http://www.vim.org/scripts/script.php?script_id=1658 下载 NERDTree。

    2. 解压下载的压缩包 nerdtree.zip，并将所有的文件复制到 D:\Vim\vimfiles\ 目录下。

WinManager 插件

  • WinManager 插件的作用

    WinManager 插件的作用是在 VIM 编辑器的侧边显示文件目录以及缓冲区目录，装有 WinManager 插件的 VIM 外观如下图所示：

    我们可以在 VIM 中直接浏览文件目标，并进入相应的文件中，而且对于打开多个缓冲区的情况，WinManager 插件也会显示以及打开的缓冲区，便于切换多个文件的缓冲区。

  • WinManager 插件的安装

    1. 请到 VIM 的官方网站 http://www.vim.org/scripts/script.php?script_id=95 下载 WinManager。

    2. 解压下载的压缩包 winmanager.zip，并将所有的文件复制到 D:\Vim\vimfiles\ 目录下。

  • WinManager 插件的使用说明：

    1. 在 Normal 模式下，按下 mw，便可以打开以及关闭 WinManager 窗口。

    2. 在 Normal 模式下，按下 <Tab>键，便可以切换缓冲区。

    3. 在 Normal 模式下，按下 <Alt>+numb键，便可以切换到对应编号的缓冲区，例如<Alt>+2 切换到2号缓冲区。

在配置文件中加入如下代码：

" ===============pathogen 支持===============
execute pathogen#infect()


" ===============拼写检查===============
set nocp
" 字典设置
let g:cvimsyn='D:\Vim\vimfiles\CVIMSYN'
let g:spchkautonext= 1
let g:spchkdialect = "usa" 

" ===============LATEX支持===============
" 设置格式
autocmd BufNewFile,BufRead *.tex set filetype=tex
" 设置编码
autocmd FileType tex setlocal et sta fileencoding=utf-8
" 设置字体
autocmd FileType tex set guifont=Yahei\ Mono:h16
" 设置设置拼写检查
autocmd BufNewFile,BufRead *.tex set spell
set shellslash
set grepprg=grep\ -nH\ $*
" 编译与查看设置
let g:tex_flavor='latex'
let g:Tex_DefaultTargetFormat = 'pdf'
let g:Tex_ViewRule_pdf = 'SumatraPDF -reuse-instance -inverse-search "gvim -c \":RemoteOpen +\%l \%f\""' 
let g:Tex_CompileRule_pdf = 'pdflatex --synctex=-1 -src-specials -interaction=nonstopmode $*'
" let g:Tex_CompileRule_pdf = 'xelatex --synctex=-1 -src-specials -interaction=nonstopmode $*'
" 设置多次编译 以包含参考文献
let g:Tex_MultipleCompileFormats = 'dvi,pdf'
" 自动换行设置
let g:tex_indent_items=0
" 不进行符号的转义显示
let g:tex_conceal=''
" TIP: if you write your \label's as \label{fig:something}, then if you type in \ref{fig: and press <C-n> you will automatically cycle through all the figure labels. Very useful!
set iskeyword+=:
" 禁用宏
" let g:Imap_FreezeImap=1
" <F2>编译
autocmd FileType tex nmap <F2> \ll|imap <F2> <ESC>:w<CR>\ll|vmap <F2> <ESC>:w<CR>\ll
" <F3>查看
autocmd FileType tex nmap <F3> <ESC>\lv|imap <F3> <ESC>\lv|vmap <F3> <ESC>\lv
" <C-b> 黑体
autocmd FileType tex imap <C-B> bf<Tab>
" 自动补全neocomplcache
autocmd FileType tex let g:neocomplcache_enable_at_startup = 1
" 自动补全字典设置
autocmd FileType tex setlocal dictionary+=D:\Vim\vimfiles\dict\english.dict
autocmd FileType tex setlocal complete+=k 

" ===============Markdown===============
" 设置格式
autocmd BufNewFile,BufRead *.{md,mdown,mkd,mkdn,markdown,mdwn} set filetype=mkd

" ===============TXT===============
" 设置格式
autocmd BufNewFile,BufRead *.{txt} set filetype=txt
autocmd FileType txt set wrap

" ===============EasyMotion===============
let g:EasyMotion_do_mapping = 0 " Disable default mappings
" Bi-directional find motion
" Jump to anywhere you want with minimal keystrokes, with just one key binding.
" `s{char}{label}`
nmap s <Plug>(easymotion-s)
" or
" `s{char}{char}{label}`
" Need one more keystroke, but on average, it may be more comfortable.
nmap s <Plug>(easymotion-s2)
" Turn on case sensitive feature
let g:EasyMotion_smartcase = 1
" JK motions: Line motions
map <Leader>j <Plug>(easymotion-j)
map <Leader>k <Plug>(easymotion-k)

" ===============格式化文本===============
" Tabular
" 以等号对其
nmap t= :Tabularize /=<CR>
vmap t= :Tabularize /=<CR>
" 对其分号之后的部分
nmap t: :Tabularize /:\zs<CR>
vmap t: :Tabularize /:\zs<CR>
" 对其逗号之后的部分
nmap t, :Tabularize /,\zs<CR>
vmap t, :Tabularize /,\zs<CR>
